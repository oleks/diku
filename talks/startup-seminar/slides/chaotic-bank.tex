\frametitle{Example: The Chaotic Bank Society\cite{muller-et-al-2009}}

\begin{columns}

\begin{column}{0.3\textwidth}

\begin{align*}
a_0 &= e-1 \\
a_1 &= 1 \cdot a_0 - 1 \\
a_2 &= 2 \cdot a_1 - 1 \\
a_3 &= 3 \cdot a_2 - 1 \\
    &\ldots \\
a_n &= n \cdot a_{n-1} - 1 \\
\end{align*}

\end{column}

\begin{column}{0.6\textwidth}

$$\lim_{n\to\infty} a_n =
\begin{cases}
-\infty & \text{if}\quad a_0 < e-1 \\
0 & \text{if}\quad a_0 = e-1 \\
+\infty & \text{if}\quad a_0 > e-1 \\
\end{cases}$$

\end{column}
